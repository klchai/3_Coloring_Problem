\subsection{Présentation du problème}
\frame
{
  \frametitle{Définitions}

 \begin{block}{Définition}

 Un \textit{graphe} est la donnée d'un couple $(V,E)$ composé de:
 \begin{itemize}
 \item $V$ d'un ensemble de \textit{sommets} ou \textit{noeuds};
 \item $E  \subseteq \{ \{ x,y \} | (x,y)\in V^2 \wedge x \neq y \}$,  d'un ensemble d'arêtes, qui sont des paires de sommets. 
 \end{itemize} 
Un graphe \textit{orienté} possède des arêtes ayant une orientation. 
 \end{block}

 \frame
{
  \frametitle{$k$-coloration}



 \begin{block}{Problème de $k$-coloration}
 Etant donné un graphe non orienté $G = (V,E)$,  et un entier $k \geq 1$, existe-t-il une \textit{coloration} $c$ : $ V \rightarrow \{1,2, \cdots, k\}$ telle que : $\forall \{u,v\} \in E, c(u) \neq c(v)$ ? 
  \end{block}
  
  On se place dans le cas $k=3$.

\section{Formalisation}

\subsection{Formalisation en logique du premier ordre}
\frame
{
  \frametitle{Formalisation en logique du premier ordre}

\begin{block}{Notations}
Un graphe \textit{non orienté} $G = (V,E)$ : 
  \begin{itemize}
  \item $n := |V| $ le nombre de noeuds;
  \item $p := |E| $ le nombre d'arêtes;
  \end{itemize} 
\end{block}

\begin{block}{Signature}
Nous considérons la signature  $\Sigma = \{ e^{r2} , c^{r2}, n^{f0}, p^{f0}  \}$ ayant la sémantique suivante: 

  \begin{itemize}
  \item $e(x,y)$ : $x$ est voisin de $y$;
  \item $c(x,j)$ : $x$ a la couleur $j$.
  \end{itemize}
 \end{block}

}

\frame
{
  \frametitle{Formalisation en logique du premier ordre}

\begin{block}{Contraintes}
  \begin{enumerate}
  \item<1-> Tout sommet du graphe est coloré;
  \item<2-> Tout sommet possède au plus une couleur;
  \item<3-> Deux sommets voisins ont des couleurs différentes. 
  \end{enumerate}
  \end{block}
  
   \begin{block}{Expressions en logique du premier ordre}

   \begin{enumerate}
  \item<1-> $\forall x  \in \llbracket 1, n \rrbracket, \exists j \in \llbracket 1,k \rrbracket , c(x,j) $;
  \item<2-> $\forall x \in \llbracket 1,n \rrbracket, \exists i \in \llbracket 1,k \rrbracket,  (c(x,i) \implies (\bigwedge_{j\neq i} \neg c(x,j))) $;
  \item<3-> $\forall x,y \in \llbracket n \rrbracket, \forall  i \in \llbracket 1,k \rrbracket , \neg (e(x,y) \wedge (c(x,i) \wedge c(y,i)) $. 
  \end{enumerate}
\end{block}
}

\section{Modèle pour SAT-Solver}

\subsection{Variables et clauses}
\frame
{
  \frametitle{Format et variables}

   \begin{block}{Fichier d'entrée}
  \begin{itemize}
  \item $n$ et $p$ sur la première ligne
  \item $p$ lignes décrivant les paires de sommets reliées par une arête
  \end{itemize}
  \end{block}
  
   \begin{block}{Variables}
  \begin{itemize}
  \item $3n$ variables $x_{(i-1) n+(j-1)}$ où $i \in \llbracket 1,n \rrbracket $, $j \in \llbracket 1,3 \rrbracket$  \end{itemize}
  \end{block}

  \begin{block}{Nombre de clauses}
  \begin{itemize}
  \item $27n + 3p$ clauses \end{itemize}
  \end{block}
}
